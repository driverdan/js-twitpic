TwitPic Javascript API Examples

media/show

var twitpic = new TwitPic();
twitpic.media.show({'id':'28ymi7'}, function (media) {
  $("#media_show").html(twitpic.thumb(media, 'thumb') + " ");
});

users/show

var twitpic = new TwitPic();
twitpic.users.show({'username':'meltingice'}, function (user) {
  for(var i=0; i < 5; i++) {
    $("#users_show").append(twitpic.thumb(user.images[i], 'thumb') + " ");
  }
});
